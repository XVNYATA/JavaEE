2、删除公告
  //为删除按钮绑定事件
  $("#deletebut").click(function(){
	  //获取被选中的checkbox
	  var boxes = $("input[id^='box_']:checked");
	  if(boxes.length==0){
		  alert("请选择需要删除的数据！")
	  }else{
		  if(window.confirm("您确认删除该记录吗?")){
			  //获取被选中的公告的id值
			  var ids = new Array();
			  $.each(boxes,function(i,item){
				  ids.push(item.value);
			  })
			  //发送请求删除记录
			  window.location = "${ctx}/notice/deleteNotice.action?page.pageIndex=${page.pageIndex}&notice.title=${notice.title}&ids="+ids.join(",");
		  }
	  }
  })
