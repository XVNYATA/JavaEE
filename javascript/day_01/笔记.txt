7、 函数定义的三种方式
   第一种定义函数的方式
  function  test(arg1,arg2,..., argN){   //arg1,arg2,..., argN形参列表
     // 函数的执行体
  }

  语法：
  function 函数的引用(形参列表){
     // 函数的执行体
  }

 形参可以零值N个

  第二种定义函数的方式  
  var 函数的引用 = function(arg1, arg2,..., argN){
     // 函数的执行体
  }
  语法：
  var 函数的引用 = function (形参列表){
     // 函数的执行体
  }

  // 匿名函数  
  (function(arg1, arg2,..., argN){
     // 函数的执行体
  })(50);



  语法：
  (function(形参列表){
   // 函数的执行体
  })(实参列表);


  
  第三种定义函数的方式
  var arg1 = "num1";
  var arg2 = "num2";
  var str = "alert(num1 * num2);";
  var test2 = new Function(arg1, arg2, str);
 
  //函数的调用
  test2(2,3);

  语法：
  var 函数的引用 = new Function(arg1, arg2, ..., argN);
  ar1-argN-1 : 形参列表
  argN : 函数的执行体。

   
8、函数调用的三种方式
     //定义一个函数
     var test = function (num1, num2){
       alert((num1 + num2));
     }    
     //第一种调用函数的方式
     window.test(10,20);  // window：javascript中最顶级的对象，可以省略

  

     // 第二种调用函数的方式
     test.call(window, 30,40);

     // 第三种调用函数的方式
     test.apply(window, [100,200]);
     test(500,600);
